using Innova.Insurance.Interfaces;
using Innova.Insurance.Interfaces.Repositories;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Innova.Insurance.Data.Repositories
{
    public class BankBranchesDataService : IBankBranchesDataService
    {
        readonly InsuranceEntities context;
        public BankBranchesDataService(InsuranceEntities ctx)
        {
            context = ctx;
        }

        public BankBranches Select(Guid ID)
        {
            var bankbranch = context.BankBranches.AsQueryable().SingleOrDefault(x => x.BankBranchID == ID);
            return bankbranch;
        }
        public List<BankBranches> GetAll()
        {
            var queryable = context.BankBranches.AsNoTracking().ToList();
            return queryable.OrderBy(x => x.Name).ToList();
        }


        public List<BankBranches> SelectByBank(Guid BankID)
        {
            var results = context.BankBranches.Where(c => c.BankID == BankID).ToList();

            return results;
        }

        public Guid Create(BankBranches CreateModel)
        {
            var item = context.BankBranches.FirstOrDefault(c => c.BankBranchID == CreateModel.BankBranchID);
            if (item == null)
            {
                if (CreateModel.BankBranchID == Guid.Empty)
                    CreateModel.BankBranchID = Guid.NewGuid();

                item = CreateModel;

                context.BankBranches.Add(item);
            }
            else
                throw new InvalidOperationException("A bank branch with the provided ID already exists");

            context.SaveChanges();

            return item.BankBranchID;
        }

        public Guid Update(BankBranches UpdateModel)
        {
            var item = context.BankBranches.FirstOrDefault(c => c.BankBranchID == UpdateModel.BankBranchID);
            if (item == null)
            {
                throw new InvalidOperationException("Could not find a bank branch matching ");
            }
            else
            {
                item.Active = UpdateModel.Active;
                item.AlternatePhone = UpdateModel.AlternatePhone;
                item.Code = UpdateModel.Code;
                item.CountryID = UpdateModel.CountryID;
                item.DateModified = DateTime.Now;
                item.EditUserID = UpdateModel.EditUserID;
                item.Email = UpdateModel.Email;
                item.Fax = UpdateModel.Fax;
                item.IPAddress = UpdateModel.IPAddress;
                item.LogonAccount = UpdateModel.LogonAccount;
                item.MainPhone = UpdateModel.MainPhone;                
                item.Name = UpdateModel.Name;
                item.Notes = UpdateModel.Notes;
                item.PhysicalAddress = UpdateModel.PhysicalAddress;
                item.PostalAddress = UpdateModel.PostalAddress;
                item.SwiftCode = UpdateModel.SwiftCode;
                item.Website = UpdateModel.Website;
            }

            context.SaveChanges();
            return item.BankBranchID;
        }



        public void Delete(Guid ID)
        {
            BankBranches bankBranches = context.BankBranches.FirstOrDefault(c => c.BankBranchID == ID);
            context.BankBranches.Remove(bankBranches);
            context.SaveChanges();
        }
    }
}
